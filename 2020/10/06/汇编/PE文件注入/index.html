PE文件注入

首先开始窝们不得不提一下PE

PE 的意思就是 Portable Executable（可移植的执行体）。它是 Win32环境自身所带的执行体文件格式。它的一些特性继承自 Unix的 Coff (common object file format)文件格式。”portable executable”（可移植的执行体）意味着此文件格式是跨win32平台的 : 即使Windows运行在非Intel的CPU上，任何win32平台的PE装载器都能识别和使用该文件格式。当然，移植到不同的CPU上PE执行体必然得有一些改变。所有 win32执行体 (除了VxD和16位的Dll)都使用PE文件格式，包括NT的内核模式驱动程序（kernel mode drivers）。因而研究PE文件格式给了我们洞悉Windows结构的良机。

这些东西都是由编译器编译的时候做好了的。

而窝们现在做的就是把已经编译好的 .exe文件加入窝们想要加入的代码。那么窝们就不得不对PE进行解析。

直接上一段代码来进行说明，加入的代码是Win32程序的MessageBoxA() 。

接下来是函数实现

使用到目录路径的，窝都用Path代替了，比如要使用ReadPEFile(Path, pFileSize)。

MESSAGEBOX的宏是用静态分析工具找到的函数地址。直接定义为宏方便使用。

function.cpp

#include"function.h"
LPVOID ReadPEFile(LPSTR lpszFile, PDWORD pFileSize)	//LPVOID是一个指向任何类型的指	
{
	FILE* pFile = NULL;
	DWORD fileSize = 0;
	LPVOID pFileBuffer = NULL;

	//打开文件	
	pFile = fopen(lpszFile, "rb");
	if (!pFile)
	{
		printf(" 无法打开 EXE 文件! ");
		return NULL;
	}
	//读取文件大小		
	fseek(pFile, 0, SEEK_END);
	fileSize = ftell(pFile);
	*pFileSize = fileSize;
	fseek(pFile, 0, SEEK_SET);
	//分配缓冲区	
	pFileBuffer = malloc(fileSize);

	if (!pFileBuffer)
	{
		printf(" 分配空间失败! ");
		fclose(pFile);
		return NULL;
	}
	/*将文件数据读取到缓冲区
	ize_t是标准C库中定义的，在64位系统中为long long unsigned int，
	非64位系统中为long unsigned int。*/
	size_t n = fread(pFileBuffer, fileSize, 1, pFile);
	if (!n)
	{
		printf(" 读取数据失败! ");
		free(pFileBuffer);
		fclose(pFile);
		return NULL;
	}
	//关闭文件	
	fclose(pFile);
	return pFileBuffer;
}

LPVOID ImageBuffer(LPVOID pFileBuffer) {
	LPVOID pImageBuffer = NULL;
	PIMAGE_DOS_HEADER pDosHeader = NULL;
	PIMAGE_NT_HEADERS pNTHeader = NULL;
	PIMAGE_FILE_HEADER pPEHeader = NULL;
	PIMAGE_OPTIONAL_HEADER32 pOptionalHeader = NULL;
	PIMAGE_SECTION_HEADER pSectionHeader = NULL;
	//验证文件是否打开
	if (!pFileBuffer)
	{
		printf("文件打开失败");
		free(pFileBuffer);
		return NULL;
	}
	//验证MZ标志
	if ((*(PWORD)pFileBuffer) != IMAGE_DOS_SIGNATURE) {
		printf("不是有效的MZ标志");
		free(pFileBuffer);
		return NULL;
	}
	//验证PE标志
	pDosHeader = (PIMAGE_DOS_HEADER)pFileBuffer;
	if ((*(PDWORD)((DWORD)pFileBuffer + pDosHeader->e_lfanew)) != IMAGE_NT_SIGNATURE) {
		printf("不是有效的PE标志");
		free(pFileBuffer);
		return NULL;
	}
	//找到NT头以后找标准PE头和可选PE头
	pNTHeader = (PIMAGE_NT_HEADERS)((DWORD)pFileBuffer + pDosHeader->e_lfanew);
	pPEHeader = (PIMAGE_FILE_HEADER)((DWORD)pNTHeader + 4);
	pOptionalHeader = (PIMAGE_OPTIONAL_HEADER32)((DWORD)pPEHeader + IMAGE_SIZEOF_FILE_HEADER);
	//获取内存中的PE大小并做初始化
	pImageBuffer = malloc(pOptionalHeader->SizeOfImage);
	memset(pImageBuffer, 0, pOptionalHeader->SizeOfImage);
	//开始内存复制
	//首先复制sizeOfHeads
	memcpy(pImageBuffer, pFileBuffer, pOptionalHeader->SizeOfHeaders);
	//剩下的就是复制节表了
	for (size_t i = 0; i < pPEHeader->NumberOfSections; i++) {
		pSectionHeader = (PIMAGE_SECTION_HEADER)((DWORD)pOptionalHeader +
			pPEHeader->SizeOfOptionalHeader + i * 40);
		memcpy((PWORD)((DWORD)pImageBuffer + pSectionHeader->VirtualAddress),
			(PWORD)((DWORD)pFileBuffer + pSectionHeader->PointerToRawData),
			pSectionHeader->SizeOfRawData);
	}
	return pImageBuffer;
}

size_t addCode(LPVOID pImageBuffer)
{
	PIMAGE_DOS_HEADER pDosHeader = NULL;
	PIMAGE_NT_HEADERS pNTHeader = NULL;
	PIMAGE_FILE_HEADER pPEHeader = NULL;
	PIMAGE_OPTIONAL_HEADER32 pOptionalHeader = NULL;
	PIMAGE_SECTION_HEADER pSectionHeader = NULL;
	if (!pImageBuffer)
	{
		printf("文件读取失败\n");
		return -1;
	}
	//判断是否是有效的MZ标志	
	if (*((PWORD)pImageBuffer) != IMAGE_DOS_SIGNATURE)//#define IMAGE_DOS_SIGNATURE 0x5A4D
	{
		printf("不是有效的MZ标志\n");
		free(pImageBuffer);
		return -1;
	}
	pDosHeader = (PIMAGE_DOS_HEADER)pImageBuffer;
	//判断是否是有效的PE标志	
	if (*((PDWORD)((DWORD)pImageBuffer + pDosHeader->e_lfanew)) != IMAGE_NT_SIGNATURE)
		//#define IMAGE_NT_SIGNATURE 0x00004550
	{
		printf("不是有效的PE标志\n");
		free(pImageBuffer);
		return -1;
	}
	pNTHeader = (PIMAGE_NT_HEADERS)((DWORD)pImageBuffer + pDosHeader->e_lfanew);
	pPEHeader = (PIMAGE_FILE_HEADER)(((DWORD)pNTHeader) + 4);//WORD Machine
	pOptionalHeader = (PIMAGE_OPTIONAL_HEADER32)((DWORD)pPEHeader + IMAGE_SIZEOF_FILE_HEADER);
	//接下来就是加入代码了MessageBox
	char shellCode[18] = {
		0x6A,0,0x6A,0,0x6A,0,0x6A,0,
		0xE8,0,0,0,0,
		0xE9,0,0,0,0
	};
	size_t shellCode_len = 18;
	PBYTE codeBegin = NULL;
	size_t i = 0;
	for (; i < pPEHeader->NumberOfSections; i++)
	{
		pSectionHeader = (PIMAGE_SECTION_HEADER)((DWORD)pOptionalHeader +
			pPEHeader->SizeOfOptionalHeader + i * 40);
		if(((pSectionHeader->SizeOfRawData)-
            ((pSectionHeader->Misc.VirtualSize)))>shellCode_len){
			//代码开始处，并完成注入
			codeBegin = (PBYTE)((DWORD)pImageBuffer + pSectionHeader->VirtualAddress + (pSectionHeader->Misc.VirtualSize));
			memcpy(codeBegin, shellCode, shellCode_len);
			break;
		}
	}
	/*判断是否找到了能加入代码的空白区*/
	if (codeBegin != NULL) {
		//1.修改E8,计算跳转的地址
		size_t imageOffset =pOptionalHeader->ImageBase +
			(((DWORD)codeBegin+ 13) - (DWORD)pImageBuffer);//计算在内存中的偏移量
		DWORD callAddr = MESSAGEBOX - imageOffset;
		*((PDWORD)(codeBegin + 9)) = callAddr;
		printf("1.通过");
		//2.修改E9,计算跳转地址,跳回entryPoint
		DWORD jmpAddr = (pOptionalHeader->ImageBase + pOptionalHeader->AddressOfEntryPoint)
			- (pOptionalHeader->ImageBase + (((DWORD)codeBegin + shellCode_len) - (DWORD)pImageBuffer));
		*(PDWORD)(codeBegin + 14) = jmpAddr;
		printf("2.通过");
		/*3.修改entryPoint,跳到内存模型中的codeBegin的地址
		entryPoint是一个相对的虚拟地址*/
		pOptionalHeader->AddressOfEntryPoint = ((DWORD)codeBegin - (DWORD)pImageBuffer);
		printf("加入代码完成");
		return i;
	}
	printf("没有足够的空间加入代码");
	return -1;
}

VOID ImageToFile(LPVOID pFileBuffer, LPVOID pImageBuffer, size_t i)
{
	PIMAGE_DOS_HEADER pDosHeader = NULL;
	PIMAGE_NT_HEADERS pNTHeader = NULL;
	PIMAGE_FILE_HEADER pPEHeader = NULL;
	PIMAGE_OPTIONAL_HEADER32 pOptionalHeader = NULL;
	PIMAGE_SECTION_HEADER pSectionHeader = NULL;
	if (!pFileBuffer||!pImageBuffer)
	{
		printf("文件读取失败\n");
		free(pFileBuffer);
		free(pImageBuffer);
		return;
	}

	//判断是否是有效的MZ标志	
	if (*((PWORD)pFileBuffer) != IMAGE_DOS_SIGNATURE)//#define IMAGE_DOS_SIGNATURE 0x5A4D
	{
		printf("不是有效的MZ标志\n");
		free(pFileBuffer);
		free(pImageBuffer);
		return;
	}
	pDosHeader = (PIMAGE_DOS_HEADER)pFileBuffer;
	//判断是否是有效的PE标志	
	if (*((PDWORD)((DWORD)pFileBuffer + pDosHeader->e_lfanew)) != IMAGE_NT_SIGNATURE)
		//#define IMAGE_NT_SIGNATURE 0x00004550
	{
		printf("不是有效的PE标志\n");
		free(pFileBuffer);
		free(pImageBuffer);
		return;
	}
	pNTHeader = (PIMAGE_NT_HEADERS)((DWORD)pFileBuffer + pDosHeader->e_lfanew);
	pPEHeader = (PIMAGE_FILE_HEADER)(((DWORD)pNTHeader) + 4);//WORD Machine
	pOptionalHeader = (PIMAGE_OPTIONAL_HEADER32)((DWORD)pPEHeader + IMAGE_SIZEOF_FILE_HEADER);
	//找到对应的节表
	pSectionHeader= pSectionHeader = (PIMAGE_SECTION_HEADER)((DWORD)pOptionalHeader +
		pPEHeader->SizeOfOptionalHeader + i * 40);
	//将改节表的数据进行复制即可
	memcpy((PWORD)((DWORD)pFileBuffer + pSectionHeader->PointerToRawData),
		(PWORD)((DWORD)pImageBuffer + pSectionHeader->VirtualAddress),
		pSectionHeader->SizeOfRawData);
	//还有最重要的一步就是要改entryPoint
	memcpy(pFileBuffer, pImageBuffer, pOptionalHeader->SizeOfHeaders);
	return VOID();
}

窝这里花了不少的时间把其中遇到的坑说一说

在计算变量jmpAddr时一开始习惯性地使用了宏DWORD是unsigned long，然而实际上jmpAddr计算出来的结果一定是负数，所以更换成signed long，当然不论是long还是int都是可以的，加一个signed起到提醒的作用。

后来经过实验发现其实是没有关系的，用宏DWORD完全没有问题，存储的数据是一样的，区别只在于是否把它看成无符号数，这里我们是直接复制数据，没有涉及到数与数之间的运算，所以是不影响程序运行的。

把注入代码的节表复制回pFileBuffer时，忘记修改pFileBuffer里的程序入口entryPoint。

需要特别注意的是，E8和E9后面填写的地址都是 .exe文件被PE Loader到内存里时的虚拟地址，与我们在程序中malloc的内存地址并不相同。

test.cpp

#include"function.h"
using namespace std;

int main()
{
	/*1.读取文件*/
	DWORD pFileSize = NULL;
	LPVOID pFileBuffer = ReadPEFile(Path,&pFileSize);
	/*2.内存格式拉伸*/
	LPVOID pImageBuffer = ImageBuffer(pFileBuffer);
	/*3.代码注入*/
	size_t i = addCode(pImageBuffer);
	/*4.返回至文件格式*/
	ImageToFile(pFileBuffer, pImageBuffer, i);
	/*5.文件存盘*/
	FileToExe(pFileBuffer, pFileSize);
    return 0;
}
