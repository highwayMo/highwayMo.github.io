图的邻接矩阵存储及遍历

题目描述

采用邻接矩阵表示无向图，完成图的创建、图的深度优先遍历、图的广度优先遍历操作。其中图的顶点信息是字符型，图中顶点序号按字符顺序排列。本输入样例中所用的图如下所示：

输入描述

第一行输入两个值，第一个是图中顶点的个数，第二个是图中边的条数 第二行输入各顶点的信息，即输入每个顶点字符 第三行开始输入每条边，每条边的形式为两个顶点的序号，中间以空格隔开，输入完一条边换行

输出描述

首先输出图的顶点信息，输出完毕换行 接着输出图的邻接矩阵，假如图中有n个顶点，则输出形式为n行n列的邻接矩阵，输出完毕换行 接下来一行输出从图的第一个顶点开始进行深度优先遍历的序列，中间以空格隔开，输出完毕换行 最后一行输出从图的第一个顶点开始进行广度优先遍历的序列，中间以空格隔开，输出完毕换行

输入样例

5 7

A B C D E

0 1 0 2 0 3 1 2 1 3 2 4 3 4

输出样例

A B C D E

0 1 1 1 0

1 0 1 1 0

1 1 0 0 1

1 1 0 0 1

0 0 1 1 0

A B C E D

A B C D E

	visited[v] = 1;

因为不是每一个图都是连通的，有些可能根本没有相连，所以需要查看visited数组来决定遍历的源节点。上面那样的写法就是默认**visited[0]**当做遍历的源节点了。

void DFSTraverse() { 
    for (size_t i = 0; i < vertexNum; i++)
    {
        if (visited[i] == 0) {
            DFSTraverse(i);
        }
    }
}

在外层套一个循环遍历visited数组，寻找遍历的源节点。

完整代码

##include <iostream>
##include <queue>
using namespace std;
const int MaxSize = 10;

class MGraph
{
public:
	MGraph(char a[], int n, int e);
	void DFSTraverse() { 
		for (size_t i = 0; i < vertexNum; i++)
		{
			if (visited[i] == 0) {
				DFSTraverse(i);
			}
		}
	}
	void BFSTraverse() { 
		for (size_t i = 0; i < vertexNum; i++)
		{
			if (visited[i] == 0) {
				BFSTraverse(i);
			}
		}
	}
	void ClearVisited();
	void Display();//输出顶点值，再输出邻接矩阵
private:
	//char vertex[MaxSize];//顶点数组
	//int arc[MaxSize][MaxSize];
	//int visited[MaxSize];
	char vertex[MaxSize] = {0};//顶点数组
	int arc[MaxSize][MaxSize] = { 0 };
	int visited[MaxSize] = { 0 };
	int vertexNum, arcNum;
	void DFSTraverse(int v);
	void BFSTraverse(int v);

};

MGraph::MGraph(char a[], int n, int e)
{
	/*ToDo
	构造函数
	初始化邻接矩阵 参数依次为各顶点信息、顶点个数、边的条数*/
	vertexNum = n;
	arcNum = e;
	int v1, v2;
	/*

		}
	}
	//初始化visited
	for (int i = 0; i < vertexNum; i++) {
		visited[i] = 0;
	}*/

	//初始化顶点
	for (int i = 0; i < vertexNum; i++) {
		vertex[i] = a[i];
	}
	//邻接矩阵赋值

		cin >> v1 >> v2;
		arc[v1][v2] = 1;
		arc[v2][v1] = 1;
	}
}

void MGraph::DFSTraverse(int v)
{
	//ToDo 以v为顶点 进行深度优先遍历
	cout << vertex[v] << " ";
	visited[v] = 1;
	for (int j = 0; j < vertexNum; j++) {
		if (arc[v][j] == 1 && visited[j] == 0)
			DFSTraverse(j);
	}
}

void MGraph::BFSTraverse(int v)
{
	//ToDo 以v为顶点 进行广度优先遍历
	int v1, v2;
	queue<int> Q;
	cout << vertex[v] << " ";
	visited[v] = 1;
	Q.push(v);
	while (!Q.empty())
	{
		int v = Q.front();//返回第一个元素的引用
		Q.pop();//删除第一个元素
		for (int j = 0; j < vertexNum; j++) {
			if (arc[v][j] == 1 && visited[j] == 0) {
				cout << vertex[j] << " ";
				visited[j] = 1;
				Q.push(j);
			}
		}
	}
}

void MGraph::ClearVisited()
{
	//ToDo 清除记录访问的visited数组
	for (size_t i = 0; i < vertexNum; i++)
	{
		visited[i] = 0;
	}
}

void MGraph::Display()
{
	//ToDo 打印顶点值，邻接矩阵
	for (size_t i = 0; i < vertexNum; i++)
	{
		cout << vertex[i] << " ";
	}
	cout << endl;

		}
		cout << endl;
	}
}

图的邻接表存储结构及遍历

题目描述

用邻接表表示有向图，完成图的创建、图的深度优先遍历、图的广度优先遍历操作。其中图的顶点信息是字符型，图中顶点序号按字符顺序排列，边的输入按照边的顶点序号从小到大的顺序排列，如下图的边的输入顺序为0 1，0 2，0 3，1 2，1 3，2 4，3 4共七条边，邻接表的边结点采用头插法。本输入样例中所用的图如下所示：

输入描述

第一行输入两个值，第一个是图中顶点的个数，第二个是图中边的条数 第二行输入各顶点的信息，即输入每个顶点字符 第三行开始输入每条边，每条边的形式为两个顶点的序号，中间以空格隔开，输入完一条边换行

输出描述

首先输出图的顶点信息，输出完毕换行 接着输出图的邻接表，格式为首先输出第一个顶点，接着输出该顶点的所有的临界点的序号，换行，然后输出下一个顶点及邻接点，以此类推 接下来一行输出从图的第一个顶点开始进行深度优先遍历的序列，中间以空格隔开，输出完毕换行 最后一行输出从图的第一个顶点开始进行广度优先遍历的序列，中间以空格隔开，输出完毕换行

输入样例

5 7

A B C D E

0 1 0 2 0 3 1 2 1 3 2 4 3 4

输出样例

A B C D E

A 3 2 1

B 3 2

C 4

D 4

E

A D E C B

A D C B E

完整代码

##include <iostream>
##include <string>
##include <queue>
using namespace std;

struct Arcnode
{
	int adjvex;
	Arcnode* next;
};

struct Vertexnode
{
	char vertex;
	Arcnode* firstedge;
};

const int Maxsize = 20;

class ALgraph
{
public:
	ALgraph(char a[], int n, int e);
	void DFSTraverse();
	void BFSTraverse();
	void Show();
	void Clearvisited();

private:
	void DFSTraverse(int v);
	void BFSTraverse(int v);

	Vertexnode adjlist[Maxsize] = {0};//顶点数组
	int visited[Maxsize] = {0};
	int vertexnum;
	int arcnum;
};

ALgraph::ALgraph(char a[], int n, int e)
{
	/*ToDo
	   构造函数
	   初始化邻接矩阵 参数依次为各顶点信息、顶点个数、边的条数*/
	vertexnum = n;
	arcnum = e;
	Arcnode* pArcNode;
	int v1, v2;
	for (size_t i = 0; i < vertexnum; i++)
	{
		adjlist[i].vertex = a[i];
		adjlist[i].firstedge = NULL;
	}
	for (size_t i = 0; i < arcnum; i++)
	{
		cin >> v1 >> v2;
		pArcNode = new Arcnode;
		pArcNode->adjvex = v2;
		//用头插法插入表头
		pArcNode->next = adjlist[v1].firstedge;
		adjlist[v1].firstedge = pArcNode;
	}
}

void ALgraph::DFSTraverse()
{
	int i;
	for (i = 0; i < vertexnum; i++)
	{
		if (visited[i] == 0)
		{
			DFSTraverse(i);
		}
	}
}

void ALgraph::DFSTraverse(int v)
{
	//ToDo  以v为顶点 进行深度优先遍历
	int j;
	Arcnode* p = NULL;
	cout << adjlist[v].vertex << " ";
	visited[v] = 1;
	p = adjlist[v].firstedge;
	while (p)
	{
		if (visited[p->adjvex] == 0) {
			DFSTraverse(p->adjvex);
		}
		p = p->next;
	}
}

void ALgraph::BFSTraverse()
{
	int i;
	for (i = 0; i < vertexnum; i++)
	{
		if (visited[i] == 0)
		{
			BFSTraverse(i);
		}
	}
}
