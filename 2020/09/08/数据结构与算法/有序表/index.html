1.求两个集合的并集和交集

这题我之前的博客其实写过了，所以稍微简单地改了改就写上了。

题目描述：

已知A和B均是由整型数据组成的集合，使用顺序表表示集合，设计算法求集合A、B的交集和并集，功能包括输入集合A，输入集合B，求A和B的并集，求A和B的交集。本题中, 线性表的第一个元素位置为1，线性表的最大长度为20。

输入描述：

各个命令以及相关数据的输入格式如下： 输入集合A：A，接下来的一行是要输入的集合元素个数n，下面是n行数据，每行数据有一个值，代表集合元素值 输入集合B：B，接下来的一行是要输入的集合元素个数n，下面是n行数据，每行数据有一个值。当输入的命令为U时，输出A和B两个集合的并集，当输入的命令为I时，输出A和B两个集合的交集，当输入的命令为E时，程序结束。注意，所有的元素均占一行。

#include<iostream>
using namespace std;
int main()
{
	int a[20],b[20],c[20],d[40],la=0,lb=0,lc=0,ld=0;
	char sel;

		if(sel=='B')
		{

			//TODO
            cin>>lb;
            for(int i=0;i<lb;i++){
                cin>>b[i];
            }
		}
		if(sel=='U')
		{
			//如果读入U，则输出集合A和B的并集
			//TODO            
            //把集合A存入并集D中
            for(int i=0;i<la;i++){
                d[i]=a[i];
                ld++;
            }
            //集合B与并集D遍历比较
            for(int i=0;i<lb;i++)//外层循环逐一取出集合B的元素
            {
                int flag=1;//设置flag标志
                for(int j=0;j<ld;j++){//内层循环用取出的元素
                                    //与交集D中所有元素比较
                    if(b[i]==d[j]) {
                        flag=0; //发现重复，flag变为0
                    }           //跳入continue
                }
                if(flag==1){ //没有发现重复，该元素存入并集D
                    d[ld]=b[i];

                }else{
                    continue;
                }
            }

                cout<<d[i]<<endl;

		    }

                cout<<c[i]<<endl;
            }
        }

}

2.在顺序表中删除值为x和y的元素

描述

顺序表的数据元素的类型为整型，初始输入若干个数据元素，可以有相同值，删去值为x和y的数据元素。

输入描述：

输入初始数据元素的个数n，占一行 输入n个初始数据元素，以空格隔开，占一行 输入x 和y，以空格隔开

输出描述：

输出删去x和y后的数据元素，以空格隔开，最后一个元素后面有空格；若删除后顺序表为空，则输出“顺序表已为空”

这题的输入已经被固定为2个了，所以难度也下降了不少。

#include<iostream>
using namespace std;
int main()
{
	int n;
    cin>>n;
    int length=n;
    int SeqList[n];
    for(int i =0;i<n;i++){
        cin>>SeqList[i];
    }
    int x,y;
    cin>>x;
    cin>>y;
    int xk=0;
    for (int i = 0;i <= length;i++)
    {
        if (SeqList[i] == x)
            xk++;
        else
            SeqList[i - xk] = SeqList[i];
    }
    length -= xk;
    int yk=0;
    for (int i = 0;i <= length;i++)
    {
        if (SeqList[i] == y)
            yk++;
        else
            SeqList[i - yk] = SeqList[i];
    }
    length -= yk;
    if(length!=0){
        for(int i=0;i<length;i++){
            cout<<SeqList[i]<<" ";
        }
    }else{
        cout<<"顺序表已为空";
    }
}

4.合并顺序表

分别输入顺序表的个数，然后再依次输入非递减有序的==顺序表==，最后将其合并，要求合并后的顺序表依然非递减有序。（注：非递减有序是指顺序表中任意一个元素的值都大于等于其前驱（如果存在））。

#include <iostream>
using namespace std;

class Seqlist {
private:
	int date[50];
	int length;
public:
	Seqlist(int a[], int n);
	~Seqlist() {}
	void insert(int x);
	void show();
};
Seqlist::Seqlist(int a[], int n) {
	for (int i = 0; i < n; i++) {
		date[i] = a[i];
	}
	/*write your code */
	length = n;
}

void Seqlist::insert(int x) {
	int i = 0;
	while (i < length && date[i] <= x) {
		i++;
	}   ///这里要注意i<length，不然最后一个数字如果相同的话就会出错
		   ///把x之前的都遍历下
	if (i == length) {
		date[i] = x;
		length++;
	}
	else {
        /*上面while循环最大能得到i==length
        *也就是说j可能等于i，无法进入for
        *所以需要上面的if来判断i是否等于length
        */
		for (int j = length; j > i; j--) {
			date[j] = date[j-1];
			length++;
		}
		/*write your code */
		date[i] = x;
	}
}

void Seqlist::show() {
	cout << length << endl;   //输出合并后的顺序表元素个数
	cout << date[0];         //输出合并后的顺序表
	for (int i = 1; i < length; i++)
		cout << " " << date[i];
	cout << endl;
}

int main()
{
	int i, n, la, lb, a[50], b[50];
	cin >> la;     //请输入第一个顺序表元素个数
	for (i = 0; i < la; i++)   //请输入第一个顺序表
		cin >> a[i];
	cin >> lb;   //请输入第二个顺序表元素个数
	Seqlist pa(a, la);

		cin >> b[i];
	for (i = 0; i < lb; i++)
		pa.insert(b[i]);
	pa.show();
	return 0;
}

2.删除单链表前驱结点

描述

假设在长度大于1的循环单链表中，具有3个结点，数据域值为1，2，3，该循环链表既无头结点也无头指针，p为指向该链表中某个结点的指针，编写一个算法删除该结点的前驱结点并从p开始的结点依次输出。

需要注意的是这一题里的单链表是==循环链表==。

#include<iostream>
using namespace std;
struct node {
	int data;
	node* next;
};
int main() {
	node a, b, c;
	a.data = 1;          //建立具有3个结点，数据域值为1，2，3的单循环链表
	a.next = &b;
	b.data = 2;
	b.next = &c;
	c.data = 3;
	c.next = &a;
	node* p = &b;
	node* q = p;
	while (q->next->next != p) {   //删除p结点的前驱
		q = q->next;
	}
	q->next = p;
	q = p;
	do {
		cout << q->data << endl;
		q = q->next;
	} while (q != p);
	return 0;
}

3.单链表的就地逆置

描述

完成单链表就地逆置算法。就地逆置只利用单链表上的结点，不再开辟新的存储空间，完成算法。（参考头插法建立单链表的算法）

这一题的算法参考了1篇博客。

先用一个指针指向first节点，然后剩下的链表再用逆置算法就好了，最后用first节点指向原来的尾节点就行。

#include <iostream> 
using namespace std;
struct Node
{
	int data;
	Node* next;
};
class LinkList
{
public:

	LinkList(int a[], int n);//尾插法在单链表中插入n个数据
	~LinkList();
	void Reverse();
	void PrintList();
private:

	Node* first;
};

LinkList::LinkList(int a[], int n)
{
	first = new Node; //附设头结点
	//Todo:尾插法在单链表中插入n个数据
	first->next = NULL;
	Node* ptemp = first;
	for (size_t i = 0; i < n; i++)
	{
		Node* node = new Node;
		node->data = a[i];
		node->next = NULL;
		ptemp->next = node;
		ptemp = node;
	}
}

void LinkList::PrintList()
{

	Node* p = first->next;
	while (p)
	{
		cout << p->data << endl;
		p = p->next;
	}
}

LinkList::~LinkList()
{

	Node* p = first, * s;
	while (p)
	{
		s = p;
		p = p->next;
		delete s;
	}
}

void LinkList::Reverse()
{
	//Todo:完成单链表就地逆置
	Node* pr;
	Node* q = NULL;
	Node* p = NULL;
	p = first->next;
	q = NULL;
	first->next = NULL;
	while (p) {
		pr = p->next;
		p->next = q;
		q = p;
		p = pr;
	}
	first->next = q;
}

4.两个有序单链表的合并

描述

将两个非递减有序单链表合并成一个非递减有序单链表；

输入样例：

5

1 3 5 7 9

6

2 3 4 6 8 10

输出样例：

1 2 3 3 4 5 6 7 8 9 10

#include <iostream> 
using namespace std;
struct Node
{
	int data;
	Node* next;
};
class LinkList
{
public:
	LinkList(int a[], int n);//尾插法在单链表中插入n个数据
	~LinkList();
	void PrintList();
	Node* GetFirst() { return first; }

private:
	Node* first;
};

LinkList::LinkList(int a[], int n)
{
	first = new Node; //附设头结点
	Node* rear = first;
	int i;
	for (i = 0; i < n; i++)
	{
		Node* s = new Node;
		s->data = a[i];
		rear->next = s;
		rear = s;
	}
	rear->next = NULL;
}

void LinkList::PrintList()
{
	Node* p = first->next;
	while (p)
	{
		cout << p->data << " ";
		p = p->next;
	}
}

LinkList::~LinkList()
{
	Node* p = first, * s;
	while (!p)
	{
		s = p;
		p = p->next;
		delete s;
	}
}

void Merge(LinkList& A, LinkList& B)
{
	//Todo:将A和B两个非递减有序单链表合并，结果放到A链表中
	Node* la = A.GetFirst();
	Node* l = A.GetFirst();
	Node* lb = B.GetFirst();
	//移动指针
	Node* Ap = la->next;
	Node* Bp = lb->next;
	la->next = NULL;
	while (Ap != NULL && Bp != NULL)
	{
		/*
		将两个结点的数据进行比较，数据较小的结点接在头结点后面，
		*/
		if (Ap->data < Bp->data)
		{
            /*如果A->data小于Bp->data
            * Ap指针向后移
            */
			la->next = Ap;
			la = Ap;
			Ap = Ap->next;
		}
		else
		{
			la->next = Bp;
			la = Bp;
			Bp = Bp->next;
		}
	}
	//若其中一个链表的结点已经全接在新表中则将另一个链表的剩余结点接在新表的后面
	if (Ap)
	{
		la->next = Ap;
	}
	if (Bp)
	{
		la->next = Bp;
	}
}

int main()
{
	int m, i, x;
	cin >> m;
	int a[3];
	for (i = 0; i < m; i++)
	{
		cin >> a[i];
	}
	LinkList la(a, m);

		cin >> b[i];
	}

	return 0;
}

//Todo:单链表插入操作
void LinkList::Insert(int i,int x)
{
    Node *p = first;
    int k = 1;
    while(p && k < i){
        p = p->next;
        k++;
    }
    Node *s = new Node;
    s->data = x;
    s->next = p->next;
    p->next = s;
    length++;
}

//Todo:单链表按值查找操作
int LinkList::Search(int x)
{
    Node *p = first->next;
    int k = 1;
    while(p && p->data != x){
        p = p->next;
        k++;
    }
    if(!p){
        return -1;
    }
    return k;
}

int LinkList::Get(int i)
{
    Node *p=first->next;
    int j=1;
    while(p&&j<i){
        p=p->next;
        j++;
    }
    if(!p)
        return 0;
    else
        return p->data;
}
